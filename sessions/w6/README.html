Functions in R

Syntax

name.of.function <- function(params) {

}

func1 <- function() {
    r <- sqrt(7)
    pi * r^2
}

func1()
## [1] 21.99115

What happens if I want to change the values?

func2 <- function(radius) {
    pi * radius^2
}

func2(10)
## [1] 314.1593

What happens if I dont pass any value?

#func2()
func3 <- function(radius = 7) {
    pi * radius^2
}

func3() ## picks default
## [1] 153.938
func3(8) ## uses the passed value
## [1] 201.0619
Returning values

implicitly

add.nos <- function(a,b) {
    a + b
}

ret.val1 <- add.nos(5,8)
ret.val1
## [1] 13
ret.val2 <- add.nos(15,20)
ret.val2
## [1] 35

explicitly using return statement

key.func <- function(a = 10, b = 50, c = 20, key = 1) {
    
    if(key == 1) {
        ret.val <- a + b
    }
    else {
        ret.val <- b + c
    }
    
    return(ret.val)
}

key.func()
## [1] 60
key.func(key=0)
## [1] 70

Control Structures

  1. if

    if(TRUE) {
    do something
}
func.test <- function(mean = 10) {
    ret.val <- 0
    
    if(mean > 5) {
        ret.val <- 20    
    }
    return(ret.val)
}

func.test()
## [1] 20
func.test(mean = 1)
## [1] 0

  1. if-else

    if(TRUE) {
    do something
} else {
    do something else
}
    if(7 > 10) {
            print("Are you serious?")
    } else {
            print("Yeah, thought so.")
    }
## [1] "Yeah, thought so."

  1. ifelse

    ifelse(CONDITION, IF TRUE, IF FALSE)
values <- c(0.5,1,-1)

values <- ifelse(values > 0.5, 1, -1)

table(values)
## values
## -1  1 
##  2  1

Use it with something cool

values <- ifelse(sample(0:1,10000,replace = T) > 0.5, 1, -1)

plot(cumsum(values), type = "l", col = "red")
lines(c(0,10000),c(0,0))
